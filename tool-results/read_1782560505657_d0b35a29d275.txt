    42→/** Build a country with minimal seed data + universal fallbacks. */
    43→function simpleCountry(
    44→  code: string, name: string, flag: string, locale: string, currency: string,
    45→  capital: string, weatherCity?: string, arabicName?: string,
    46→  payments?: PaymentMethod[], transport?: TransportMethod[],
    47→): CountryInfo {
    48→  return {
    49→    code, name, flag, locale, currency, capital, arabicName,
    50→    majorCities: [capital],
    51→    newsSources: [{ name: `${name} National News`, handle: `@${code.toLowerCase()}_news`, type: "media" as const }],
    52→    localBrands: [],
    53→    landmarks: [],
    54→    weatherCity: weatherCity || capital,
    55→    paymentMethods: payments && payments.length ? payments : UNIVERSAL_PAYMENTS,
    56→    transportMethods: transport && transport.length ? transport : UNIVERSAL_TRANSPORT,
    57→  };
    58→}

    66→}
    67→function t(id: string, name: string, type: TransportMethod["type"], icon: string, description: string): TransportMethod {
    68→  return { id, name, type, icon, description };
    69→}
